群组-协议
IChatManagerGroup

//创建群组
//群组设置
EMGroupStyleSetting *setting = [[EMGroupStyleSetting alloc] init];
//群组最大人数
setting.groupMaxUsersCount = maxUsersCount;
//群组类型
 eGroupStyle_PrivateOnlyOwnerInvite 私有群组，只能owner权限的人邀请人加入
 eGroupStyle_PrivateMemberCanInvite 私有群组，owner和member权限的人可以邀请人加入
 eGroupStyle_PublicJoinNeedApproval 公开群组，允许非群组成员申请加入，需要管理员同意才能真正加入该群组
 eGroupStyle_PublicOpenJoin         公开群组，允许非群组成员加入，不需要管理员同意
 eGroupStyle_PublicAnonymous        公开匿名群组，允许非群组成员加入，不需要管理员同意
setting.groupStyle = eGroupStyle_PrivateMemberCanInvite;
